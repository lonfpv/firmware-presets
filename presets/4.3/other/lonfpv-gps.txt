#$ TITLE: LonFPV enable GPS settings
#$ FIRMWARE_VERSION: 4.3
#$ CATEGORY: OTHER
#$ STATUS: COMMUNITY
#$ KEYWORDS: defaults, reset, GPS
#$ AUTHOR: LonFPV (Rich Blomseth)
#$ DESCRIPTION: Resets the GPS settings to LonFPV 4.3 defaults 

# -- LonFPV GPS feature customizations --
feature GPS

# -- LonFPV GPS beeper customizations --
beeper -ARMING_GPS_FIX
beeper -GPS_STATUS

# -- LonFPV GPS hardware customizations --
set mag_hardware = AUTO
set baro_hardware = AUTO

# -- LonFPV GPS aux channel customizations --
aux 8 46 1 1425 1575 0 0

# -- LonFPV GPS receiver customizations --
set gps_provider = UBLOX
set gps_sbas_mode = WAAS
set gps_rescue_allow_arming_without_fix = ON

# -- LonFPV GPS OSD customizations --
set osd_gps_speed_pos = 2400
set osd_gps_lon_pos = 2464
set osd_gps_lat_pos = 2432
set osd_gps_sats_pos = 2048
set osd_home_dir_pos = 2080
set osd_home_dist_pos = 2112
