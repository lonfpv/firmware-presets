# -- LonFPV common settings --
#$ INCLUDE: presets/4.3/other/lonfpv_common.txt

# -- Manufacturer: HappyModel --
manufacturer_id HAMO

# -- DMA --
# ADC 1: DMA2 Stream 0 Channel 0
dma ADC 1 0
dma pin A09 NONE
dma pin A10 NONE

# -- Features --
feature -ANTI_GRAVITY

# -- Serial Ports --
serial 20 1 115200 57600 0 115200
serial 0 0 115200 57600 0 115200
# VTX smart audio -- redundant with HDZero VTX preset
serial 1 1 115200 57600 0 115200

# -- Master --
# from RC6
set crsf_use_negotiated_baud = OFF
set vbat_scale = 112
set ibata_scale = 285
set vtx_band = 1
set vtx_channel = 1
set vtx_power = 1
set gyro_1_sensor_align = CW0FLIP
set gyro_1_align_pitch = 1800
set gyro_1_align_yaw = 0

# -- Overclock (can help the lower performance F411; default is 108MHz)
# cpu_overclock set to 120MHz

# -- Motor Protocol --
# DSHOT600 required for 8K PID loop; DSHOT150 may be sufficient for 2K PID loop
set motor_pwm_protocol = DSHOT300
set dshot_burst = OFF
set dshot_idle_value = 450
set dshot_bidir = ON
set dshot_bitbang = AUTO
set motor_poles = 12
# Denominator for Gyro Update Frequency / PID Loop Frequency (4 for 2K)
set pid_process_denom = 4
